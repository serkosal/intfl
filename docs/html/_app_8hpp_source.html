90 int waitInput()
91 {
92 std::wstring cmd = L"";
93 #ifdef USE_N_CURSES
94 wint_t wch;
95 auto res = wget_wch(M_cmdsWin.getPtr(), &wch);
96
97 while (wch != WEOF && wch != L'\n' && wch != L'\r')
98 {
99 if (res == KEY_CODE_YES)
100 {
101 if (wch == KEY_ENTER || wch == KEY_END)
102 {
103 cmd = L"";
104 break;
105 }
106 else if (wch == KEY_BACKSPACE && cmd.size())
107 {
108 cmd.resize(cmd.size() - 1);
109 M_cmdsWin.printcr(cmd);
110 }
111 }
112 else if (res == ERR)
113 {
114 cmd = L"";
115 return 1;
116 }
117 else if (res == OK && !std::iswcntrl(wch))
118 {
119 cmd += static_cast<wchar_t>(wch);
120 wchar_t buf[2] = {static_cast<wchar_t>(wch), L'\0'};
121 M_cmdsWin.printr(std::wstring(buf));
122 }
123
124 res = wget_wch(M_cmdsWin.getPtr(), &wch);
125 }
126 M_cmdsWin.clear();
127 M_cmdsWin.refresh();
128 #else
129 auto &res = std::getline(std::wcin, cmd);
130 if (res.fail())
131 { return 1; }
132 #endif
133
134 return 0;
135 }
